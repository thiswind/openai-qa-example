    def bubble_sort(arr):    
        n = len(arr)
        
        # 在range(n)中的每一个i，都不断检查arr[i]和arr[i+1]
        for i in range(n):
            # 如果arr[i]>arr[i+1]，那么就交换它们的位置
            for j in range(0, n-i-1):
                if arr[j] > arr[j+1]:
                    arr[j], arr[j+1] = arr[j+1], arr[j]
                    
    arr = [14,46,43,27,57,41,45,21,70]
    bubble_sort(arr)
    
    print ("排序后的数组:")
    for i in range(len(arr)):
        print ("%d" %arr[i]),
